## Platform-specific Installation

<CodeTabs
  tabs={{
    "Python": `# Using pip
pip install Snapjobs

# Using poetry

poetry add Snapjobs-ai

# Using conda

conda install -c Snapjobs Snapjobs-ai`,
    "Node.js": `# Using npm
npm install @Snapjobs/client

# Using yarn

yarn add @Snapjobs/client

# Using pnpm

pnpm add @Snapjobs/client`,
    "Docker": `# Pull the Snapjobs server image
docker pull Snapjobs/server:latest

# Run with environment variables

docker run -d \\
-e Snapjobs_API_KEY=your_api_key \\
-e Snapjobs_ENVIRONMENT=production \\
-p 3000:3000 \\
Snapjobs/server`
}}
/>

## SDK Initialization

<CodeTabs
  tabs={{
    "Python": `from install Snapjobs import MatchingEngine
import os

# Using environment variables

engine = MatchingEngine(
api_key=os.getenv('Snapjobs_API_KEY'),
environment='production'
)

# Or using direct configuration

engine = MatchingEngine(
api_key='your_api_key',
environment='production',
timeout=30000
)`,
    "Node.js": `import { SnapjobsClient } from '@Snapjobs/client';

// Using environment variables
const client = new SnapjobsClient({
apiKey: process.env.Snapjobs_API_KEY
});

// Or using direct configuration
const client = new SnapjobsClient({
apiKey: 'your_api_key',
environment: 'production',
timeout: 30000
});`
}}
/>

## Common Issues

<div className="space-y-6 mt-8">
  <FeaturedExample
    title="API Key Issues"
    description="Troubleshoot API key configuration problems"
    code={`// Check if API key is properly set
if (!process.env.Snapjobs_API_KEY) {
  throw new Error('Snapjobs_API_KEY not found in environment');
}

// Verify API key format
if (!/^Snapjobs\_[a-zA-Z0-9]{32}$/.test(process.env.Snapjobs_API_KEY)) {
throw new Error('Invalid API key format');
}`}
/>

  <FeaturedExample
    title="Connection Issues"
    description="Handle common connection problems"

  await client.testConnection();
} catch (error) {
  if (error.code === 'ECONNREFUSED') {
    console.error('Connection refused. Check your network.');
  } else if (error.code === 'TIMEOUT') {
    console.error('Connection timed out. Try increasing timeout.');
  }
}`}

</div>
