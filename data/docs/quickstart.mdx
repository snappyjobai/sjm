## Installation

<CodeTabs
  tabs={{
    pip: `pip install sjm`,
    npm: `npm install @sjm/client`,
    yarn: `yarn add @sjm/client`,
  }}
/>

## Basic Usage

<CodeTabs
  tabs={{
    "Python": `from sjm import MatchingEngine, Project

# Initialize the engine

engine = MatchingEngine()

# Create a project

project = Project(
id="p1",
description="Need a React Native developer",
required_skills=["React Native", "JavaScript"],
budget_range=(1000, 5000),
complexity="medium",
timeline=30
)

# Find matches

matches = engine.get_top_matches(project)

# Print top matches

for match in matches:
print(f"Match score: {match['combined_score']}")
print(f"Freelancer: {match['freelancer'].name}")`,

    "JavaScript": `import { SJMClient } from '@sjm/client';

// Initialize client
const client = new SJMClient({ apiKey: 'your_api_key' });

// Find matches
const matches = await client.match({
description: "Need a React Native developer",
requiredSkills: ["React Native", "JavaScript"],
budget: {
min: 1000,
max: 5000
}
});

console.log(matches);`,

    "REST": `curl -X POST https://api.sjm.dev/v1/match \\

-H "Authorization: Bearer YOUR_API_KEY" \\
-H "Content-Type: application/json" \\
-d '{
"description": "Need a React Native developer",
"requiredSkills": ["React Native", "JavaScript"],
"budget": {
"min": 1000,
"max": 5000
}
}'`
}}
/>
