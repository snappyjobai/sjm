## Basic Integration

<FeaturedExample
  title="Simple Matching Query"
  description="Use the SDK to find matching freelancers."
  code={{
    node: `import { SnapjobsClient } from '@Snapjobs/client-sdk';

const client = new SnapjobsClient({ apiKey: process.env.Snapjobs_API_KEY });
const matches = await client.match({
projectDescription: "Build a React Native app",
requiredSkills: ["React Native", "JavaScript"],
budget: { min: 1000, max: 5000 }
});
console.log(matches);`,
    python: `from Snapjobs_client import Client

client = Client(api_key=os.getenv("Snapjobs_API_KEY"))
matches = client.match(
project_description="Build a React Native app",
required_skills=["React Native", "JavaScript"],
budget={"min": 1000, "max": 5000}
)
print(matches)`,
    rest: `curl -X POST https://api.Snapjobs.dev/v1/match \\
-H "Authorization: Bearer YOUR_API_KEY" \\
-H "Content-Type: application/json" \\
-d '{
"projectDescription": "Build a React Native app",
"requiredSkills": ["React Native", "JavaScript"],
"budget": { "min": 1000, "max": 5000 }
}'`
}}
/>
