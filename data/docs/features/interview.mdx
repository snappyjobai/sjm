## Interview Generation

<CodeTabs
  tabs={{
    Python: `from sjm.ai import InterviewEngine

engine = InterviewEngine()

interview = engine.generate_interview({
"role": "Senior React Developer",
"required_skills": ["React", "TypeScript", "Redux"],
"experience_level": "senior",
"interview_duration": 60,
"question_types": ["technical", "behavioral", "system_design"]
})

# Access questions

for q in interview.questions:
print("Question:", q.text)
print("Rubric:", q.rubric)`,

    NodeJS: `import { InterviewEngine } from '@sjm/client'

const engine = new InterviewEngine()

const interview = await engine.generateInterview({
role: "Senior React Developer",
requiredSkills: ["React", "TypeScript", "Redux"],
experienceLevel: "senior",
interviewDuration: 60,
questionTypes: ["technical", "behavioral", "system_design"]
})

// Access questions
interview.questions.forEach(q => {
console.log("Question:", q.text)
console.log("Rubric:", q.rubric)
})`
}}
/>
